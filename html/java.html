Operatori

Java supporta diversi tipi di operatori per eseguire operazioni su variabili e valori:

Java 
// Operatori aritmetici

// Classe base (superclasse)
public class Animale {
    protected String nome;
    
    public Animale(String nome) {
        this.nome = nome;
    }
    
    public void mangia() {
        System.out.println(nome + " sta mangiando");
    }

// Classe derivata (sottoclasse)
public class Cane extends Animale {
    private String razza;
    
    public Cane(String nome, String razza) {
        super(nome);  // Chiama il costruttore della superclasse
        this.razza = razza;
    }
    
    public void abbaia() {
        System.out.println(nome + " fa: Bau Bau!");
    }
}

// Utilizzo
Cane rex = new Cane("Rex", "Labrador");
rex.mangia();   // Metodo ereditato
rex.abbaia();   // Metodo proprio
